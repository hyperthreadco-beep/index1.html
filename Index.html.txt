/* ══════════════════════════════════════════
   LOADING SCREEN INTRO
══════════════════════════════════════════ */
(function(){
  const loader=document.getElementById('loader');
  const logEl=document.getElementById('loader-log');
  const bar=document.getElementById('loader-bar');
  const pctEl=document.getElementById('loader-pct');
  const done=document.getElementById('loader-done');

  const steps=[
    {text:'> INITIALIZING SYSTEM...',pct:18},
    {text:'> LOADING PORTFOLIO ASSETS...',pct:42},
    {text:'> ESTABLISHING NETWORK LINK...',pct:67},
    {text:'> CALIBRATING HUD INTERFACE...',pct:88},
    {text:'> BOOT SEQUENCE COMPLETE.',pct:100},
  ];

  let stepIdx=0;

  function typeLine(text,pct,cb){
    const div=document.createElement('div');
    logEl.appendChild(div);
    let i=0;
    const id=setInterval(()=>{
      div.textContent=text.slice(0,++i);
      if(i>=text.length){
        clearInterval(id);
        bar.style.width=pct+'%';
        pctEl.textContent=pct+'%';
        setTimeout(cb,280);
      }
    },22);
  }

  function runStep(){
    if(stepIdx>=steps.length){
      done.classList.add('show');
      setTimeout(()=>{
        loader.classList.add('out');
        setTimeout(()=>loader.remove(),900);
      },750);
      return;
    }
    const s=steps[stepIdx++];
    typeLine(s.text,s.pct,runStep);
  }

  setTimeout(runStep,500);
})();

/* ══════════════════════════════════════════
   HUD BACKGROUND CANVAS
══════════════════════════════════════════ */
(function(){
  const canvas=document.getElementById('hud-canvas');
  const ctx=canvas.getContext('2d');
  let W,H;

  function resize(){W=canvas.width=innerWidth;H=canvas.height=innerHeight;}
  window.addEventListener('resize',resize);
  resize();

  /* Corner bracket helper */
  function bracket(x,y,sx,sy,sz){
    ctx.beginPath();
    ctx.moveTo(x+sx*sz,y);
    ctx.lineTo(x,y);
    ctx.lineTo(x,y+sy*sz);
    ctx.stroke();
  }

  /* Floating reticles */
  const reticles=Array.from({length:6},()=>({
    x:Math.random()*innerWidth,
    y:Math.random()*innerHeight,
    r:18+Math.random()*28,
    vx:(Math.random()-.5)*.18,
    vy:(Math.random()-.5)*.18,
    phase:Math.random()*Math.PI*2,
  }));

  /* Sparse grid annotation points */
  const gridPts=[];
  for(let gx=.15;gx<1;gx+=.28){
    for(let gy=.18;gy<1;gy+=.32){
      gridPts.push({fx:gx,fy:gy,off:Math.random()*Math.PI*2});
    }
  }

  /* Data annotation panels — fixed screen positions */
  const panels=[
    {fx:.06,fy:.22,lines:['ALT  003.2m','BRG  047°','SPD  0.0kt']},
    {fx:.84,fy:.68,lines:['LAT  24.86N','LNG  067.01E','ACC  ±2m']},
    {fx:.06,fy:.76,lines:['SYS  NOMINAL','NET  LINK OK','CPU  12%']},
    {fx:.78,fy:.18,lines:['TGT  LOCKED','RNG  —.—km','STA  ACTIVE']},
  ];

  let t=0;

  function draw(){
    ctx.clearRect(0,0,W,H);
    t+=.005;

    /* ── CORNER BRACKETS ── */
    const bsz=44;
    ctx.strokeStyle='rgba(10,10,10,0.13)';
    ctx.lineWidth=1.5;
    bracket(34,34, 1, 1,bsz);
    bracket(W-34,34,-1, 1,bsz);
    bracket(34,H-34, 1,-1,bsz);
    bracket(W-34,H-34,-1,-1,bsz);

    /* corner coord labels */
    ctx.font='9px "Share Tech Mono",monospace';
    ctx.fillStyle='rgba(10,10,10,0.14)';
    ctx.textAlign='left';  ctx.fillText('X:0000 Y:0000',50,54);
    ctx.fillText('X:0000 Y:'+(H|0).toString().padStart(4,'0'),50,H-50);
    ctx.textAlign='right'; ctx.fillText('X:'+(W|0).toString().padStart(4,'0')+' Y:0000',W-50,54);
    ctx.fillText('X:'+(W|0).toString().padStart(4,'0')+' Y:'+(H|0).toString().padStart(4,'0'),W-50,H-50);

    /* ── FLOATING RETICLES ── */
    reticles.forEach(r=>{
      r.x+=r.vx; r.y+=r.vy;
      if(r.x<0)r.x=W; if(r.x>W)r.x=0;
      if(r.y<0)r.y=H; if(r.y>H)r.y=0;

      const pulse=.5+Math.sin(t*1.7+r.phase)*.35;
      const a=pulse*.11;

      /* outer circle */
      ctx.beginPath();
      ctx.arc(r.x,r.y,r.r,0,Math.PI*2);
      ctx.strokeStyle=`rgba(10,10,10,${a})`;
      ctx.lineWidth=.8;
      ctx.stroke();

      /* inner small circle */
      ctx.beginPath();
      ctx.arc(r.x,r.y,r.r*.35,0,Math.PI*2);
      ctx.strokeStyle=`rgba(10,10,10,${a*.6})`;
      ctx.lineWidth=.5;
      ctx.stroke();

      /* crosshair arms */
      ctx.strokeStyle=`rgba(10,10,10,${a*.7})`;
      ctx.lineWidth=.5;
      ctx.beginPath();
      ctx.moveTo(r.x-r.r*1.5,r.y); ctx.lineTo(r.x-r.r*.55,r.y);
      ctx.moveTo(r.x+r.r*.55,r.y); ctx.lineTo(r.x+r.r*1.5,r.y);
      ctx.moveTo(r.x,r.y-r.r*1.5); ctx.lineTo(r.x,r.y-r.r*.55);
      ctx.moveTo(r.x,r.y+r.r*.55); ctx.lineTo(r.x,r.y+r.r*1.5);
      ctx.stroke();

      /* center dot */
      ctx.beginPath();
      ctx.arc(r.x,r.y,1.4,0,Math.PI*2);
      ctx.fillStyle=`rgba(10,10,10,${a*2})`;
      ctx.fill();
    });

    /* ── GRID ANNOTATION DOTS ── */
    gridPts.forEach((gp,i)=>{
      const x=gp.fx*W+Math.sin(t*.35+gp.off)*4;
      const y=gp.fy*H+Math.cos(t*.28+gp.off)*4;
      const fade=(.5+Math.sin(t*.4+gp.off)*.5)*0.09;
      ctx.beginPath();
      ctx.arc(x,y,1.5,0,Math.PI*2);
      ctx.fillStyle=`rgba(10,10,10,${fade})`;
      ctx.fill();
      /* tick line */
      ctx.strokeStyle=`rgba(10,10,10,${fade*.6})`;
      ctx.lineWidth=.5;
      ctx.beginPath();
      ctx.moveTo(x+4,y); ctx.lineTo(x+16,y);
      ctx.stroke();
      ctx.font='8px "Share Tech Mono",monospace';
      ctx.textAlign='left';
      ctx.fillStyle=`rgba(10,10,10,${fade*.8})`;
      ctx.fillText(`${(gp.fx*100)|0}.${(gp.fy*100)|0}`,x+19,y+3);
    });

    /* ── FLOATING DATA PANELS ── */
    panels.forEach((p,pi)=>{
      const x=p.fx*W;
      const y=p.fy*H;
      const drift=Math.sin(t*.25+pi*1.3)*3;
      const px=x, py=y+drift;
      const fade=.12+Math.sin(t*.18+pi*.9)*.04;

      const lh=14, padX=10, padY=8;
      const boxW=118, boxH=p.lines.length*lh+padY*2;

      /* panel box */
      ctx.strokeStyle=`rgba(10,10,10,${fade*.9})`;
      ctx.lineWidth=.7;
      ctx.strokeRect(px,py,boxW,boxH);

      /* yellow top accent line */
      ctx.fillStyle=`rgba(232,255,0,${fade*1.2})`;
      ctx.fillRect(px,py,boxW,2);

      /* text lines */
      ctx.font='9px "Share Tech Mono",monospace';
      ctx.textAlign='left';
      p.lines.forEach((line,li)=>{
        /* randomise last value occasionally */
        let display=line;
        if(Math.random()<.004){
          const parts=line.split('  ');
          if(parts.length>1){
            const chars='0123456789ABCDEF';
            display=parts[0]+'  '+Array.from({length:parts[1].length},()=>chars[Math.floor(Math.random()*chars.length)]).join('');
          }
        }
        ctx.fillStyle=`rgba(10,10,10,${fade*1.6})`;
        ctx.fillText(display,px+padX,py+padY+li*lh+8);
      });
    });

    /* ── SLOW HORIZONTAL SCAN LINE ── */
    const scanY=((Math.sin(t*.22)+1)*.5)*H;
    const grad=ctx.createLinearGradient(0,scanY,W,scanY);
    grad.addColorStop(0,'transparent');
    grad.addColorStop(.4,`rgba(232,255,0,0.04)`);
    grad.addColorStop(.6,`rgba(232,255,0,0.04)`);
    grad.addColorStop(1,'transparent');
    ctx.beginPath();
    ctx.moveTo(0,scanY); ctx.lineTo(W,scanY);
    ctx.strokeStyle=grad;
    ctx.lineWidth=1;
    ctx.stroke();

    requestAnimationFrame(draw);
  }
  draw();
})();

/* ══════════════════════════════════════════
   PARTICLE CANVAS — floating tech nodes
══════════════════════════════════════════ */
(function(){
  const canvas=document.getElementById('particle-canvas');
  const ctx=canvas.getContext('2d');
  let W,H,particles=[];
  const N=55,LINK=120;

  function resize(){
    W=canvas.width=innerWidth;
    H=canvas.height=innerHeight;
  }
  window.addEventListener('resize',resize);
  resize();

  function rnd(a,b){return a+Math.random()*(b-a)}

  for(let i=0;i<N;i++){
    particles.push({
      x:rnd(0,W),y:rnd(0,H),
      vx:rnd(-.3,.3),vy:rnd(-.3,.3),
      r:rnd(.8,2.2)
    });
  }

  function draw(){
    ctx.clearRect(0,0,W,H);
    // links
    for(let i=0;i<N;i++){
      for(let j=i+1;j<N;j++){
        const p=particles[i],q=particles[j];
        const d=Math.hypot(p.x-q.x,p.y-q.y);
        if(d<LINK){
          ctx.beginPath();
          ctx.moveTo(p.x,p.y);ctx.lineTo(q.x,q.y);
          ctx.strokeStyle=`rgba(10,10,10,${(1-d/LINK)*0.08})`;
          ctx.lineWidth=.6;
          ctx.stroke();
        }
      }
    }
    // dots
    particles.forEach(p=>{
      ctx.beginPath();
      ctx.arc(p.x,p.y,p.r,0,Math.PI*2);
      ctx.fillStyle='rgba(10,10,10,0.18)';
      ctx.fill();
    });
    // update
    particles.forEach(p=>{
      p.x+=p.vx;p.y+=p.vy;
      if(p.x<0)p.x=W;if(p.x>W)p.x=0;
      if(p.y<0)p.y=H;if(p.y>H)p.y=0;
    });
    requestAnimationFrame(draw);
  }
  draw();
})();

/* ══════════════════════════════════════════
   TITLE SCRAMBLE — on section entry
══════════════════════════════════════════ */
const CHARS='ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789#@&%';
function scramble(el){
  const original=el.textContent.trim();
  let frame=0,total=18;
  const id=setInterval(()=>{
    el.textContent=original.split('').map((c,i)=>{
      if(c===' ')return ' ';
      if(frame/total > i/original.length) return c;
      return CHARS[Math.floor(Math.random()*CHARS.length)];
    }).join('');
    if(++frame>total){el.textContent=original;clearInterval(id);}
  },40);
}
